function updateWinterVisuals() { const winter = document.getElementById('winter'); // --- DYNAMIC SCALING LOGIC --- // Top boundary (minY) is 150, Bottom boundary (maxY) is 680. // We want 150y to be 0.5 scale and 680y to be 1.0 scale. const minY = 150; const maxY = 680; // Calculate percentage of vertical progress (0 at top, 1 at bottom) const travelPercent = (state.winterY - minY) / (maxY - minY); // Apply linear interpolation: 0.5 + (0.5 * percentage) const scale = 0.5 + (0.5 * travelPercent); // Update position and scale // We adjust the offset (62 and 160) by the scale so she stays grounded winter.style.left = (state.winterX - (62 * scale)) + 'px'; winter.style.top = (state.winterY - (160 * scale)) + 'px'; // Apply horizontal flip if walking left, while maintaining the dynamic scale const direction = state.keys["KeyA"] ? -1 : (state.keys["KeyD"] ? 1 : (winter.style.transform.includes("scaleX(-1)") ? -1 : 1)); winter.style.transform = `scale(${scale}) scaleX(${direction})`; } function updateMovement() { let dx = 0, dy = 0; if (state.keys["KeyW"]) dy -= 4; if (state.keys["KeyS"]) dy += 4; if (state.keys["KeyA"]) dx -= 4; if (state.keys["KeyD"]) dx += 4; if (dx !== 0 || dy !== 0) { state.winterX = Math.max(50, Math.min(1230, state.winterX + dx)); state.winterY = Math.max(150, Math.min(680, state.winterY + dy)); // Update frame animation frameCount++; const winter = document.getElementById('winter'); if (frameCount % 10 === 0) { winter.src = (winter.src.includes("Idle")) ? "WinterWalk1.png" : "WinterIdle.png"; } } else { document.getElementById('winter').src = "WinterIdle.png"; } updateWinterVisuals(); }
